import API from '../../utils/request';
 
/**
 * Teacher - Gets usage summary
 *
 * @param sessionId
 * @param userId
 */
export const getUsageSummaryDataForTeacher = (sessionId, userId) =>
  API.get('/SlmsTeacher', {
    params: {
      command: 'get_usage',
      sid: sessionId,
      user_id: userId,
    },
  }).then(response => response.output_data[0].Teacher[0]);
